#!/bin/sh
install_aarch_box() {  # <- INSTALAR BOX ARM64 # 01
   ARQUITETURA=$(dpkg --print-architecture)
   [ "$ARQUITETURA" = "arm64" ] || [ "$ARQUITETURA" = "aarch64" ] || return 1

# Aqui você já está dentro do Debian rodando /bin/sh como root

#

sudo dpkg --add-architecture armhf
sudo apt-get update
sudo apt-get install -y git cmake cabextract gcc-arm-linux-gnueabihf libc6-dev-armhf-cross || { echo "Falha ao instalar dependências de compilação"; return 1; }
sudo apt install -y libc6:armhf || { echo "Falha ao instalar dependências de compilação"; return 1; }

# --- LINK PARA O LINKER ---
sudo ln -sf /usr/arm-linux-gnueabihf/lib/ /usr/arm-linux-gnueabihf/lib/arm-linux-gnueabihf

# --- COMPILAR BOX86 E INSTALAR ---
if [ ! -f ~/box86/build/box86 ]; then
rm -rf box86
cd ~
git clone https://github.com/ptitSeb/box86 || echo "Git box86 falhou."
cd box86 && { rm -rf build && mkdir build && cd build; } || return 1
git checkout v0.3.8 || { echo "Tag v0.3.8 não encontrada"; return 1; }
VERSION="gcc-cross"

cmake .. \
  -DCMAKE_C_COMPILER=arm-linux-gnueabihf-gcc \
  -DARM_DYNAREC=ON \
  -DCMAKE_BUILD_TYPE=RelWithDebInfo \
  -DCMAKE_C_FLAGS="-I/usr/arm-linux-gnueabihf/include" \
  -DCMAKE_EXE_LINKER_FLAGS="-L/usr/arm-linux-gnueabihf/lib -L/usr/lib/${VERSION}/arm-linux-gnueabihf/14" \
  -DCMAKE_C_COMPILER_WORKS=1

make -j$(nproc) || return 1
sudo make install
# sudo systemctl restart systemd-binfmt (não tem systemd)

elif [ -f ~/box86/build/box86 ]; then
echo "Compilação de box86 já feita! INSTALANDO.."
cd ~/box86/build/ && echo "Entrou $PWD"
make -j$(nproc) || return 1
sudo make install
fi

if [ ! -f ~/box64/build/box64 ]; then
cd ~
git clone https://github.com/ptitSeb/box64 || echo "Git box64 falhou."
cd box64 && { rm -rf build && mkdir build && cd build; } || return 1
cmake .. -DARM_DYNAREC=ON -DCMAKE_BUILD_TYPE=RelWithDebInfo

make -j3 || return 1
sudo make install
# sudo systemctl restart systemd-binfmt (não tem systemd)

elif [ -f ~/box64/build/box64 ]; then
echo "Compilação de box64 já feita! INSTALANDO.."
cd ~/box64/build/ && echo "Entrou $PWD"
make -j$(nproc) || return 1
sudo make install
fi
USEREOF
EOF
 }  # <- INSTALAR BOX ARM64 # 01 -> 01
install_aarch_box
EOFB

#!/bin/bash

download_aarch_box_local() {  # <- BAIXAR BOX PRÉ-COMPILADO ARM64 # 02 # BASH
$PREFIX/bin/bash << 'EOF'
# Configura o script para fechar imediatamente caso ocorra qualquer erro
set -e

ARQUITETURA=$(dpkg --print-architecture)

TARGET_DIR_ONE=$(ls -d $PREFIX/var/lib/proot-distro/*/debian/*/usr/local/bin | head -1)
TARGET_DIR=$(ls -d $PREFIX/var/lib/proot-distro/*/debian/*/home/teste/ | head -1)
cd $TARGET_DIR || exit 1

if [[ "$ARQUITETURA" == "arm" || "$ARQUITETURA" == "armhf" ]]; then

ls $TARGET_DIR_ONE/box86 && { echo 'box86 já está instalado, encerrando..'; exit 0; }

FILE_NAME="box_compiled_armhf.tar.gz"
SHA_NAME="box_compiled_armhf.tar.gz.sha256"
elif [[ "$ARQUITETURA" == "arm64" || "$ARQUITETURA" == "aarch64" ]]; then

ls $TARGET_DIR_ONE/box86 $TARGET_DIR_ONE/box64 &>/dev/null && { echo 'binários box86 e box64 já foram instalados, encerrando..'; exit 0; }

FILE_NAME="box_compiled.tar.gz"
SHA_NAME="box_compiled.tar.gz.sha256"
else echo "não existe um pré-compilado que eu tenha configurado para arquiteturas além de armhf e arm64 que operam em celulares. encerrando modo download.."
exit 1
fi

# Configurações de links e arquivos
REPO_URL="https://github.com/diogoeae1/type_proot_install_files/raw/main"

echo "=== [1/3] Baixando box para a distribuição PRoot e a assinatura SHA256 ==="
# Baixa o arquivo compacto e sua respectiva chave de verificação
curl --progress-bar -L --continue-at - --retry 1 --retry-delay 1 -o "$FILE_NAME" "$REPO_URL/$FILE_NAME"
curl --progress-bar -L --continue-at - --retry 1 --retry-delay 1 -o "$SHA_NAME" "$REPO_URL/$SHA_NAME"

echo "=== [2/3] Verificando a integridade do arquivo ==="
# Valida se o download bate exatamente com a chave gerada na compilação
if sha256sum -c "$SHA_NAME"; then
    echo " Verificação concluída com sucesso!"
    rm "$SHA_NAME" # Limpa a chave temporária após validação
else
    echo " Erro: O arquivo baixado está corrompido ou foi modificado!"
    exit 1
fi

echo "=== [3/3] Verifica o arquivo! PRoot ==="
ls $TARGET_DIR/$FILE_NAME && echo "=== Concluído com sucesso em $TARGET_DIR ===" || echo "Erro ao baixar $FILE_NAME para $TARGET_DIR"

EOF
} # <- INSTALAR BOX PRÉ-COMPILADO ARM64 # 02 -> 02 # BASH
download_aarch_box_local
EOFD

docking() {
# O loop pega o ID_CONTADOR exatamente de onde o Panel 1 parou e roda 9 vezes
for i in {1..9}; do
    xfconf-query -c xfce4-panel -p "/plugins/plugin-$ID_CONTADOR" -n -t string -s "showdesktop"
    adicionar_ao_painel $ID_CONTADOR
    ((ID_CONTADOR++))
echo "$ID_CONTADOR" > ~/carrinho_panel.txt # <- SALVA O ID ATUAL EM UM ARQUIVO
echo "$ID_CONTADOR"
done
}

[ ! -z "$(cat "$LISTA" | tr -d '[:space:]')" ] || { docking; echo 'A saida estava vazia ou continha apenas espacos/quebras de linha'; }

# --- LOOP DE LANÇADORES ---
while IFS= read -r DESKTOP_FILE || [ -n "$DESKTOP_FILE" ]; do
    [ -z "$DESKTOP_FILE" ] && continue
    
    xfconf-query -c xfce4-panel -p "/plugins/plugin-$ID_CONTADOR" -n -t string -s "launcher"
    xfconf-query -c xfce4-panel -p "/plugins/plugin-$ID_CONTADOR/items" -n -t string -s "$DESKTOP_FILE" -a
    
    adicionar_ao_painel $ID_CONTADOR
    ((ID_CONTADOR++))
done < "$LISTA"
