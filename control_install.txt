  if [ "$arg" = "apps:extra" ]; then echo "apps:extra"; contador=$((contador + 1)); fi

done

prosseguindo.."

# Armazena os argumentos garantindo espaços nas extremidades para a busca
args=" $* "

# Verifica se AMBOS os argumentos estão presentes na string

    *" box "* )

            *" box:local "* )
                echo "Erro. O argumento box:local não pode estar presente enquanto o argumento box está ativo."

        esac
        ;;
esac

adicionar_ao_painel() {
    local id_atual=$1
    # Adicionamos o novo ID ao texto que já existia na variável LISTA_DE_IDS
    LISTA_DE_IDS="$LISTA_DE_IDS -t int -s $id_atual"

Originalmente, usa-se o comando head -1 combinado com o grep para capturar apenas a primeira ocorrência.
