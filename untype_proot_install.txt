#!/bin/bash
ls $PREFIX/bin/bash || exit 1
apt-get update
pkg install -y x11-repo
pkg install -y termux-x11-nightly
pkg install -y wget || exit 1
pkg install -y unzip || exit 1
cd ~
wget -O type_proot_install_v1.0.2.zip https://github.com/diogoeae1/type_proot_install/releases/download/v1.0.2/type_proot_install_v1.0.2.zip
unzip type_proot_install_v1.0.2.zip
cd ~/.type_proot_install || exit 1
chmod +x *
./type_proot_install.txt || exit 1
echo 'execute ".type-help" para saber como usar isto.'
